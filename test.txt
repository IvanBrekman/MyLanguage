func SquareEquiation(a, b, c) {
    def d = b * b - 4 * a * c;

    def ans1 = -1;
    def ans2 = -1;

    if   (d <  0) return 0;
    elif (d == 0) {
        ans1 = ans2 = -b / (2 * a);

        print(ans1, ans2);

        return 1;
    }
    else {
        ans1 = (-b - sqrt(d)) / (2 * a);
        ans2 = (-b + sqrt(d)) / (2 * a);

        print(abs(ans1), abs(ans2));

        return 2;
    }

    return -666;
}

    def a = enter();

}
